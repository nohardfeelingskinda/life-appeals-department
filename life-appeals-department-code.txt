function randomItem(arr){ return arr[Math.floor(Math.random()*arr.length)]; }
function newCase(){
  const n = Math.floor(100000 + Math.random()*900000);
  const c = `NHFK-${new Date().getFullYear()}-${n}`;
  caseNo.textContent = c;
  caseNo2.textContent = c;
}
function reset(){
  newCase();
  resultBox.classList.remove('active');
  loadingBox.classList.remove('active');
  progressBar.style.width = '0%';
  loadMsg.textContent = 'Opening your file...';
  document.getElementById('appeal').focus();
}
function fileAppeal(){
  resultBox.classList.remove('active');
  loadingBox.classList.add('active');
  fileBtn.disabled = true;
  let steps = 0;
  const chosen = [...loadingMessages].sort(() => Math.random() - .5).slice(0, 6);
  const timer = setInterval(() => {
    loadMsg.textContent = chosen[steps] || 'Preparing your final decision...';
    progressBar.style.width = `${Math.min(100, (steps+1) * 17)}%`;
    steps++;
    if(steps >= 6){
      clearInterval(timer);
      setTimeout(showResult, 350);
    }
  }, 620);
}
function showResult(){
  const v = randomItem(verdicts);
  stamp.textContent = v.stamp;
  reason.textContent = v.text;
  signature.textContent = randomItem(signatures);
  loadingBox.classList.remove('active');
  resultBox.classList.add('active');
  fileBtn.disabled = false;
}
